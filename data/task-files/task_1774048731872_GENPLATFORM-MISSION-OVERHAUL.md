```javascript
// Account pool configuration
const ACCOUNTS = [
  { id: 'account1', sessionPath: '/root/.config/claude/session1.json', status: 'active', lastUsed: 0 },
  { id: 'account2', sessionPath: '/root/.config/claude/session2.json', status: 'active', lastUsed: 0 }
];

let currentAccountIndex = 0;

function getNextAccount() {
  // Round-robin selection
  const account = ACCOUNTS[currentAccountIndex];
  currentAccountIndex = (currentAccountIndex + 1) % ACCOUNTS.length;
  return account;
}

async function executeWithFallback(command, options = {}) {
  const primaryAccount = getNextAccount();
  
  try {
    return await executeWithAccount(command, primaryAccount, options);
  } catch (error) {
    if (error.message.includes('rate limit') || error.message.includes('limit exceeded')) {
      console.log(`[ACCOUNT ROTATION] ${primaryAccount.id} rate limited, switching...`);
      
      // Try other account
      const fallbackAccount = ACCOUNTS.find(a => a.id !== primaryAccount.id);
      if (fallbackAccount) {
        return await executeWithAccount(command, fallbackAccount, options);
      }
      
      // Both limited — wait 60 seconds and retry
      console.log('[ACCOUNT ROTATION] Both accounts limited, waiting 60s...');
      await sleep(60000);
      return await executeWithAccount(command, primaryAccount, options);
    }
    throw error;
  }
}

// Health check — ping accounts every 5 minutes
setInterval(async () => {
  for (const account of ACCOUNTS) {
    try {
      await checkAccountStatus(account);
      account.status = 'active';
    } catch {
      account.status = 'degraded';
      console.log(`[ACCOUNT HEALTH] ${account.id} is degraded`);
    }
  }
}, 5 * 60 * 1000);
```

## TASK-C2: Implement execution queue — web UI parity with Telegram
**Problem**: Web UI sends tasks but system stops. Telegram works fine.
**Root cause**: Web requests have timeout (30s). Long-running tasks timeout and appear stopped.
**Solution**: Queue-based execution with SSE streaming

**File**: genplatform-api/src/services/ExecutionQueue.js (create new)

```javascript
class ExecutionQueue {
  constructor() {
    this.queue = [];
    this.running = false;
    this.clients = new Map(); // SSE clients
  }
  
  addTask(taskId, command, projectId) {
    this.queue.push({ taskId, command, projectId, status: 'queued', addedAt: Date.now() });
    this.emit('task_queued', { taskId, position: this.queue.length });
    
    if (!this.running) this.processNext();
    return taskId;
  }
  
  async processNext() {
    if (this.queue.length === 0) { this.running = false; return; }
    
    this.running = true;
    const task = this.queue.shift();
    task.status = 'running';
    
    this.emit('task_started', { taskId: task.taskId });
    
    try {
      // Execute via OpenClaw/Claude
      const result = await executeWithFallback(task.command);
      
      task.status = 'completed';
      this.emit('task_completed', { taskId: task.taskId, result });
      
      // Save to progress
      await updateProgress(task.taskId, 'completed', result);
      
    } catch (error) {
      task.status = 'failed';
      this.emit('task_failed', { taskId: task.taskId, error: error.message });
      
      // Auto-retry once
      if (!task.retried) {
        task.retried = true;
        this.queue.unshift(task); // Put back at front
      }
    }
    
    // Always continue to next task
    setTimeout(() => this.processNext(), 500);
  }
  
  // SSE: subscribe to task updates
  subscribe(clientId, res) {
    this.clients.set(clientId, res);
    res.on('close', () => this.clients.delete(clientId));
  }
  
  emit(event, data) {
    const message = `data: ${JSON.stringify({ event, ...data })}\n\n`;
    this.clients.forEach(client => {
      try { client.write(message); } catch { }
    });
    
    // Also send to Telegram
    sendTelegramUpdate(`[Queue] ${event}: ${JSON.stringify(data)}`);
  }
}

const queue = new ExecutionQueue();
module.exports = { queue };
```

```typescript
  const totalChars = JSON.stringify(messages).length;
  const estimatedTokens = Math.ceil(totalChars / 4);
  const usagePercent = (estimatedTokens / 200000) * 100;
  
  if (usagePercent > 90) {
    // Hard reset
    return trimToSystemPrompt(messages);
  } else if (usagePercent > 80) {
    // Soft reset  
    return trimToLast5(messages);
  }
  
  return messages; // No trim needed
}
```
