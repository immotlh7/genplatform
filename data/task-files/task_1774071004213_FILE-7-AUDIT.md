api_test() {
  local method="$1"
  local path="$2"
  local body="$3"
  local expect="$4"
  local name="$5"

  if [ "$method" = "GET" ]; then
    RESPONSE=$(curl -s -m 10 "$BASE$path")
    STATUS=$(curl -s -o /dev/null -w "%{http_code}" -m 10 "$BASE$path")
  else
    RESPONSE=$(curl -s -m 10 -X "$method" "$BASE$path" \
      -H "Content-Type: application/json" \
      -d "$body")
    STATUS=$(curl -s -o /dev/null -w "%{http_code}" -m 10 -X "$method" "$BASE$path" \
      -H "Content-Type: application/json" \
      -d "$body")
  fi

  local result="UNKNOWN"
  if echo "$RESPONSE" | grep -q "$expect" && [ "$STATUS" != "500" ]; then
    result="PASS"
    PASS=$((PASS+1))
    echo "  PASS: $name"
    echo "- PASS: \`$method $path\` — $name" >> "$REPORT"
  elif [ "$STATUS" = "404" ] || [ "$STATUS" = "405" ]; then
    result="WARN"
    WARNINGS=$((WARNINGS+1))
    echo "  WARN: $name (HTTP $STATUS)"
    echo "- WARN: \`$method $path\` — $name (HTTP $STATUS)" >> "$REPORT"
  else
    result="FAIL"
    FAIL=$((FAIL+1))
    echo "  FAIL: $name (status=$STATUS response=${RESPONSE:0:80})"
    echo "- FAIL: \`$method $path\` — $name (HTTP $STATUS)" >> "$REPORT"
  fi
}

SEC_PASS=0
SEC_FAIL=0
SEC_WARN=0

sec_check() {
  local name="$1"
  local result="$2"   # PASS | FAIL | WARN
  local detail="$3"
  echo "- $result: $name — $detail" >> "$REPORT"
  echo "  $result: $name"
  [ "$result" = "PASS" ] && SEC_PASS=$((SEC_PASS+1))
  [ "$result" = "FAIL" ] && SEC_FAIL=$((SEC_FAIL+1))
  [ "$result" = "WARN" ] && SEC_WARN=$((SEC_WARN+1))
}

# ── Path traversal protection ──────────────────────────

TRAVERSAL=$(curl -s "http://localhost:3000/api/projects/genplatform-ai/files?path=/../../etc/passwd" | head -c 200)
if echo "$TRAVERSAL" | grep -q "root:\|error\|Error\|Invalid"; then
  sec_check "Path traversal protection" "PASS" "Blocked /../ attempts in Files API"

  sec_check "Path traversal protection" "FAIL" "Files API may allow path traversal — check the validation in files/route.ts"
